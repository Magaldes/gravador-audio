# NOTAS DE INSTALAÇÃO
# ==============================================
#
# 1. Instalar Python 3.8+:
#    Download: https://www.python.org/downloads/
#    Ou via Winget: winget install Python.Python.3.11
#
# 2. Criar ambiente virtual (recomendado):
#    python -m venv venv
#    venv\Scripts\activate
#
# 3. Instalar pacotes:
#    pip install -r requirements.txt
#
# 4. Configurar APIs (se usar transcrição):
#    - OpenAI: Obter API key em https://platform.openai.com/
#    - Google: Configurar credenciais em https://cloud.google.com/speech-to-text/
#    - Azure: Obter chave em https://portal.azure.com/
#
# ==============================================
# EXEMPLOS DE USO
# ==============================================
#
# --- OpenAI Transcrição ---
# ```python
# import openai
# openai.api_key = "sua-api-key"
# 
# with open("recordings/reunioes/arquivo.mp3", "rb") as audio:
#     transcript = openai.Audio.transcribe("whisper-1", audio)
#     print(transcript["text"])
# ```
#
# --- Google Cloud Speech ---
# ```python
# from google.cloud import speech
# 
# client = speech.SpeechClient()
# with open("recordings/reunioes/arquivo.mp3", "rb") as audio:
#     content = audio.read()
# 
# audio = speech.RecognitionAudio(content=content)
# config = speech.RecognitionConfig(
#     encoding=speech.RecognitionConfig.AudioEncoding.MP3,
#     sample_rate_hertz=16000,
#     language_code="pt-BR",
# )
# 
# response = client.recognize(config=config, audio=audio)
# for result in response.results:
#     print(result.alternatives[0].transcript)
# ```
#
# --- Azure Speech ---
# ```python
# import azure.cognitiveservices.speech as speechsdk
# 
# speech_config = speechsdk.SpeechConfig(
#     subscription="SUA_KEY",
#     region="REGIAO"
# )
# audio_input = speechsdk.AudioConfig(
#     filename="recordings/reunioes/arquivo.mp3"
# )
# recognizer = speechsdk.SpeechRecognizer(
#     speech_config=speech_config,
#     audio_config=audio_input
# )
# result = recognizer.recognize_once()
# print(result.text)
# ```
